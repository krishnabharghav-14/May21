            var a="_bh_arg_hav";

            function casechange(a){
                for(i=0;i< a.length;i++){

            Question 11:
            Create a function that returns an array of strings sorted by length in ascending order.

            var a=["a","ccc","bb","dddd"];
            var temp=null;
            function sort(a){
                for(i=0;i< a.length;i++){
                    for(j=i+1;j< a.length;j++){
                        if(a[i].length>a[j].length){
                            temp=a[i];                  //Here i have used a sorting method to sort the strings in an array in ascending order 
                            a[i]=a[j];
                            a[j]=temp;
                        }
                    }                
                }
                return a;
            }
            console.log(sort(a));


            output:
            [
            "a",
            "bb",
            "ccc",
            "dddd"
            ]
